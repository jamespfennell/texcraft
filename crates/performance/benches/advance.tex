% While loop: repeatedly executes #2 while `\ifnum #1` is true
\def\while#1#2{
  \ifnum #1
    #2%
    \while{#1}{#2}\fi
}

\def\a{\advance \i by \k}

\countdef \n 32767
\n = 1000

\countdef \i 32764
\countdef \k 1
\countdef \j 32763
\j = 0
\while{\j < \n}{
  \advance \j by 1
  %
  \i = \n
  \while{\i > 0}{
    \advance \i by -1
    \a
    \a
    \a
    \a
    \a
    \a
    \a
    \a
    \a
    \a
    \a
  }
}
