\def\swap#1#2{
  \count 80 #1
  #1 #2
  #2 \count 80 
}

\def\fibonnaci#1{
  \countdef\n 1
  \n #1
  \countdef\a 3
  \a 0
  \countdef\b 4
  \b 1
  \countdef\k 2
  \k 1

  \fibonnaciInc
}

\def\fibonnaciInc{
  \advance \a \b
  \swap \a \b
  \the\a;
  \advance \k 1
  \ifnum\k<\n\fibonnaciInc\fi
}

\fibonnaci{20}
